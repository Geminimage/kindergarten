  (function (jQ) {

        account$ = jQ("#account");

    if ( cookie_acct === undefined ){
      jQ("#rememberPwdIcon").removeClass("checkbox_checked");
    }else if(cookie_acct.length) {
      account$.attr("value", WXM.Helpers.getCookie("remember_acct"));
    }

    account$
        .focus(function () {
          loginForm.addClass("login_on_un");
        })
        .blur(function () {
          loginForm.removeClass("login_on_un");
        });
    jQ("#password")
        .focus(function () {
          loginForm.addClass("login_on_pwd");
        })
        .blur(function () {
          loginForm.removeClass("login_on_pwd");
        });
    jQ("#rememberPwd").click(function () {
      var icon = jQ("#rememberPwdIcon");
      if (icon.hasClass("checkbox_checked")) {
        icon.removeClass("checkbox_checked");
      } else {
        icon.addClass("checkbox_checked");
      }
    });


    var _loginForm = createLoginForm({
      target:'#login-form', //'<div></div>'
